#!/bin/bash

echo "üöÄ Arabiyya Pro - Quick Start Script"
echo "===================================="
echo ""

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Check if Node.js is installed
if ! command -v node &> /dev/null; then
    echo -e "${RED}‚ùå Node.js topilmadi!${NC}"
    echo "Iltimos, Node.js 18+ o'rnating: https://nodejs.org/"
    exit 1
fi

echo -e "${GREEN}‚úÖ Node.js topildi: $(node --version)${NC}"

# Check if MongoDB is running
if ! pgrep -x "mongod" > /dev/null; then
    echo -e "${YELLOW}‚ö†Ô∏è  MongoDB ishlamayapti${NC}"
    echo "MongoDB ishga tushiring yoki MongoDB Atlas ishlatishingiz mumkin"
else
    echo -e "${GREEN}‚úÖ MongoDB ishlayapti${NC}"
fi

echo ""
echo "üì¶ Dependencies o'rnatilmoqda..."
echo ""

# Install frontend dependencies
echo -e "${BLUE}Frontend dependencies...${NC}"
npm install

# Install backend dependencies
echo -e "${BLUE}Backend dependencies...${NC}"
cd backend
npm install
cd ..

echo ""
echo -e "${GREEN}‚úÖ Barcha dependencies o'rnatildi!${NC}"
echo ""

# Check if .env exists
if [ ! -f "backend/.env" ]; then
    echo -e "${YELLOW}‚ö†Ô∏è  .env fayl topilmadi${NC}"
    echo "backend/.env.example dan nusxa olinmoqda..."
    cp backend/.env.example backend/.env
    echo -e "${GREEN}‚úÖ .env fayl yaratildi${NC}"
    echo ""
    echo -e "${YELLOW}DIQQAT: backend/.env faylini tahrirlang:${NC}"
    echo "  1. MONGODB_URI - MongoDB connection string"
    echo "  2. JWT_SECRET - Random secret key"
    echo "  3. GEMINI_API_KEY - Google Gemini API key"
    echo ""
fi

echo ""
echo "üéâ Setup tugadi!"
echo ""
echo "Keyingi qadamlar:"
echo ""
echo -e "${BLUE}1. backend/.env faylini tahrirlang${NC}"
echo -e "${BLUE}2. Backend ishga tushiring:${NC}"
echo "   cd backend && npm run dev"
echo ""
echo -e "${BLUE}3. Yangi terminal ochib, Frontend ishga tushiring:${NC}"
echo "   npm run dev"
echo ""
echo -e "${GREEN}Backend:${NC}  http://localhost:5000"
echo -e "${GREEN}Frontend:${NC} http://localhost:3000"
echo ""
echo "Muvaffaqiyatlar! üöÄ"
